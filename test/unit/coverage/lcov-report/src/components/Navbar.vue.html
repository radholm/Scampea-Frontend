          <router-link to="/employees" class="navbar-item" active-class="is-active" @click.native="toggle">Employees</router-link>

<script>

  computed: {
    loggedIn () {
      return this.$store.getters.getLoginStatus
    },
    user () {
      return this.$store.getters.getUser
    }
  },
  data () {
    return {
      showNav: false

  methods: {
    toggle () {
