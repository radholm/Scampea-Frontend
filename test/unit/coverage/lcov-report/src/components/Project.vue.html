<template>
  <section class="section" v-if="ready">
 
    <nav class="breadcrumb" aria-label="breadcrumbs">
        <ul>
            <li><router-link to="/projects">Projects</router-link></li>
            <li class="is-active"><a href="#" aria-current="page">{{ project.name }}</a></li>
        </ul>
    </nav>  
 
    <h2 class="title is-4">{{ project.name }}</h2>
    <h3 class="subtitle is-6">Created at: {{ project.created_at }}</h3>
    <p>{{ timeLog }}</p>
    <ul v-for="user in projectUsers">
      <li>{{ user }}</li>
      <li>{{ user.first_name }}</li>
    </ul>   
    <hr>
 
  </section>
  <loading v-else></loading>
</template>
 
<script>
import Loading from '@/components/Loading'
 
export default {
  components: { Loading },
  props: ['id'],
  data () {
    return {
      loading: true
    }
  },
  mounted () {
    this.$store.dispatch('fetchProject', this.id)
    .then(() => {
      this.loading = false
    })
 
    this.$store.dispatch('fetchUserLogTime')
    this.$store.dispatch('fetchEmployees')
    this.$store.dispatch('fetchProjectUsers', this.id)
  },
  computed: {
    ready () {
      return this.loggedIn && !this.loading
    },
    loggedIn () {
      return this.$store.getters.getLoginStatus
    },
    project () {
      console.log(this.$store.getters.getProject)
      return this.$store.getters.getProject
    },
    projectUsers () {
      return this.$store.getters.getProjectEmployees
    },
    timeLog () {
      return this.$store.getters.getUserTimeLogs.filter(timelog => timelog.id === parseInt(this.id))
    }
  }
}
</script>
