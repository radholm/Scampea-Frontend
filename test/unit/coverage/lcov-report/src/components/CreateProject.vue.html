<script>
import AddMembers from './CreateProjectComponents/AddMembers'
const fileDefault = { name: 'Select file...' }
const srcDefault = 'http://bulma.io/images/placeholders/128x128.png'
 
export default {
  components: {AddMembers},
  name: 'settings',
  data () {
    return {
      file: fileDefault,
      competence: '',
      expertise: '',
      password: '',
      src: srcDefault
    }
  },
  computed: {
    loggedIn () {
      return this.$store.getters.getLoginStatus
    },
    pictureSrc () {
      return this.$store.getters.getUserPicture
    }
  },
  methods: {
    updateUser () {
      this.$store.dispatch('updateUser')
    },
    fileUpload (e) {
      let file = e.target.files[0]
      if (file) {
        this.file = e.target.files[0]
        this.src = window.URL.createObjectURL(e.target.files[0])
      } else {
        this.file = fileDefault
        this.src = srcDefault
      }
    }
  }
}
 
</script>
