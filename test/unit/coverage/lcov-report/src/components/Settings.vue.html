<template>
  <div class="container" v-if="loggedIn">
    <section class="section">
      <h1 class="title">Account Settings</h1>
      <h2 class="subtitle">Update your general settings</h2>
        <hr>
      <form v-on:submit.prevent="updateUser">
        <div class="field media">
          <figure class="media-left">
            <p class="image is-128x128">
              <img :src="pictureSrc">
            </p>
          </figure>
          <div class="file media-content">
            <label class="file-label">
              <input v-on:change="fileUpload" class="file-input" type="file" accept="image/png, image/jpg">
                <span class="file-cta">
                <span class="file-label">Choose an avatar</span>
              </span>
              <span class="file-name">{{ fileName }}</span>
            </label>
          </div>
        </div>
          <hr>
        <h1 class="title">Experience</h1>
        <h2 class="subtitle">Update your personal info</h2>
        <div class="field">
          <label class="label">Expertise</label>
          <div class="control">
            <textarea class="textarea" placeholder="Expertise"></textarea>
          </div>
        </div>
        <div class="field is-grouped">
          <div class="control">
            <input class="button is-primary" type="submit" value="Update Settings">
          </div>
        </div>
      </form>
        <hr>
      <h1 class="title">Change Password</h1>
      <h2 class="subtitle">Update your password</h2>
      <form v-on:submit.prevent="changePassword">
        <div class="field">
          <label class="label">Old password</label>
          <div class="control">
            <input class="input" type="password" placeholder="Old password">
          </div>
        </div>
        <div class="field">
          <label class="label">New password</label>
          <div class="control">
            <input class="input" type="password" placeholder="New Password">
          </div>
        </div>
        <div class="field">
          <label class="label">Confirm new password</label>
          <div class="control">
            <input class="input" type="password" placeholder="Confirm new password">
          </div>
        </div>
        <div class="field is-grouped">
          <div class="control">
            <input class="button is-primary" type="submit" value="Change Password">
          </div>
        </div>
      </form>
    </section>
  </div>
</template>
 
<script>
import axios from 'axios'
export default {
  name: 'settings',
  data () {
    return {
      file: '',
      expertise: '',
      password: '',
      pictureSrc: '',
      user: ''
    }
  },
  computed: {
    loggedIn () {
      return this.$store.getters.getLoginStatus
    },
    oldPictureSrc () {
      return this.$store.getters.getUserPicture
    },
    fileName () {
      return this.file.name || 'Select file...'
    }
  },
  mounted () {
    axios.get('/api/user', {
      headers: {
        Authorization: 'Bearer ' + localStorage.getItem('accessToken')
      }
    })
    .then(response => {
      console.log(response)
      this.user = response.data
    })
    .catch(error => {
      console.log(error)
    })
    this.pictureSrc = this.oldPictureSrc
  },
  methods: {
    updateUser () {
      this.$store.dispatch('updateUser')
    },
    fileUpload (e) {
      let file = e.target.files[0]
      if (file) {
        this.file = e.target.files[0]
        this.pictureSrc = window.URL.createObjectURL(e.target.files[0])
      } else {
        this.file = ''
        this.pictureSrc = this.oldPictureSrc
      }
    }
  }
}
</script>
